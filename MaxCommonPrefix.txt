#include <iostream>
#include <vector>

using namespace std;

    string longestCommonPrefix(vector<string>& strs) {
        if(strs.empty())
        {
            string empty_str;
            return empty_str;
        }
        string max_pref = strs[0];
        int end_max_pref = strs[0].size();
        for (int i = 0; i < strs.size(); ++i)
        {
            int j = 0;
            for(; j < strs[i].size() 
            && j < max_pref.size()
            && strs[i][j] == max_pref[j]; 
            ++j);
            if(j < end_max_pref)
            {
               end_max_pref = j;   
            }
            if(end_max_pref == 0)
            {
                break;
            }
        }
        string pref (max_pref.begin(), max_pref.begin() + end_max_pref);
        return pref;
    }
